class TestGUIStartup:
    """GUI 啟動測試"""
    
    @pytest.mark.skipif(not PLAYWRIGHT_AVAILABLE, reason="Playwright not installed")
    def test_gui_starts(self):
        """測試 GUI 啟動"""
        with sync_playwright() as p:
            # 啟動瀏覽器
            browser = p.chromium.launch(headless=False)  # 有頭模式，可見
            page = browser.new_page()
            
            # 導航到本地 HTML 檔案（如果有的話）
            # 或使用其他方式啟動 GUI
            
            # 截圖
            page.screenshot(path="/mnt/openwebui_data/gui_startup.png")
            
            browser.close()
